229def check_ieee_macros(config): 

230 priv = [] 

231 pub = [] 

232 

233 macros = [] 

234 

235 def _add_decl(f): 

236 priv.append(fname2def("decl_%s" % f)) 

237 pub.append('NPY_%s' % fname2def("decl_%s" % f)) 

238 

239 # XXX: hack to circumvent cpp pollution from python: python put its 

240 # config.h in the public namespace, so we have a clash for the common 

241 # functions we test. We remove every function tested by python's 

242 # autoconf, hoping their own test are correct 

243 _macros = ["isnan", "isinf", "signbit", "isfinite"] 

244 for f in _macros: 

245 py_symbol = fname2def("decl_%s" % f) 

246 already_declared = config.check_decl(py_symbol, 

247 headers=["Python.h", "math.h"]) 

248 if already_declared: 

249 if config.check_macro_true(py_symbol, 

250 headers=["Python.h", "math.h"]): 

251 pub.append('NPY_%s' % fname2def("decl_%s" % f)) 

252 else: 

253 macros.append(f) 

254 # Normally, isnan and isinf are macro (C99), but some platforms only have 

255 # func, or both func and macro version. Check for macro only, and define 

256 # replacement ones if not found. 

257 # Note: including Python.h is necessary because it modifies some math.h 

258 # definitions 

259 for f in macros: 

260 st = config.check_decl(f, headers=["Python.h", "math.h"]) 

261 if st: 

262 _add_decl(f) 

263 

264 return priv, pub

266def check_types(config_cmd, ext, build_dir): 

267 private_defines = [] 

268 public_defines = [] 

269 

270 # Expected size (in number of bytes) for each type. This is an 

271 # optimization: those are only hints, and an exhaustive search for the size 

272 # is done if the hints are wrong. 

273 expected = {'short': [2], 'int': [4], 'long': [8, 4], 

274 'float': [4], 'double': [8], 'long double': [16, 12, 8], 

275 'Py_intptr_t': [8, 4], 'PY_LONG_LONG': [8], 'long long': [8], 

276 'off_t': [8, 4]} 

277 

278 # Check we have the python header (-dev* packages on Linux) 

279 result = config_cmd.check_header('Python.h') 

280 if not result: 

281 python = 'python' 

282 if '__pypy__' in sys.builtin_module_names: 

283 python = 'pypy' 

284 raise SystemError( 

285 "Cannot compile 'Python.h'. Perhaps you need to " 

286 "install {0}-dev|{0}-devel.".format(python)) 

287 res = config_cmd.check_header("endian.h") 

288 if res: 

289 private_defines.append(('HAVE_ENDIAN_H', 1)) 

290 public_defines.append(('NPY_HAVE_ENDIAN_H', 1)) 

291 res = config_cmd.check_header("sys/endian.h") 

292 if res: 

293 private_defines.append(('HAVE_SYS_ENDIAN_H', 1)) 

294 public_defines.append(('NPY_HAVE_SYS_ENDIAN_H', 1)) 

295 

296 # Check basic types sizes 

297 for type in ('short', 'int', 'long'): 

298 res = config_cmd.check_decl("SIZEOF_%s" % sym2def(type), headers=["Python.h"]) 

299 if res: 

300 public_defines.append(('NPY_SIZEOF_%s' % sym2def(type), "SIZEOF_%s" % sym2def(type))) 

301 else: 

302 res = config_cmd.check_type_size(type, expected=expected[type]) 

303 if res >= 0: 

304 public_defines.append(('NPY_SIZEOF_%s' % sym2def(type), '%d' % res)) 

305 else: 

306 raise SystemError("Checking sizeof (%s) failed !" % type) 

307 

308 for type in ('float', 'double', 'long double'): 

309 already_declared = config_cmd.check_decl("SIZEOF_%s" % sym2def(type), 

310 headers=["Python.h"]) 

311 res = config_cmd.check_type_size(type, expected=expected[type]) 

312 if res >= 0: 

313 public_defines.append(('NPY_SIZEOF_%s' % sym2def(type), '%d' % res)) 

314 if not already_declared and not type == 'long double': 

315 private_defines.append(('SIZEOF_%s' % sym2def(type), '%d' % res)) 

316 else: 

317 raise SystemError("Checking sizeof (%s) failed !" % type) 

318 

319 # Compute size of corresponding complex type: used to check that our 

320 # definition is binary compatible with C99 complex type (check done at 

321 # build time in npy_common.h) 

322 complex_def = "struct {%s __x; %s __y;}" % (type, type) 

323 res = config_cmd.check_type_size(complex_def, 

324 expected=[2 * x for x in expected[type]]) 

325 if res >= 0: 

326 public_defines.append(('NPY_SIZEOF_COMPLEX_%s' % sym2def(type), '%d' % res)) 

327 else: 

328 raise SystemError("Checking sizeof (%s) failed !" % complex_def) 

329 

330 for type in ('Py_intptr_t', 'off_t'): 

331 res = config_cmd.check_type_size(type, headers=["Python.h"], 

332 library_dirs=[pythonlib_dir()], 

333 expected=expected[type]) 

334 

335 if res >= 0: 

336 private_defines.append(('SIZEOF_%s' % sym2def(type), '%d' % res)) 

337 public_defines.append(('NPY_SIZEOF_%s' % sym2def(type), '%d' % res)) 

338 else: 

339 raise SystemError("Checking sizeof (%s) failed !" % type) 

340 

341 # We check declaration AND type because that's how distutils does it. 

342 if config_cmd.check_decl('PY_LONG_LONG', headers=['Python.h']): 

343 res = config_cmd.check_type_size('PY_LONG_LONG', headers=['Python.h'], 

344 library_dirs=[pythonlib_dir()], 

345 expected=expected['PY_LONG_LONG']) 

346 if res >= 0: 

347 private_defines.append(('SIZEOF_%s' % sym2def('PY_LONG_LONG'), '%d' % res)) 

348 public_defines.append(('NPY_SIZEOF_%s' % sym2def('PY_LONG_LONG'), '%d' % res)) 

349 else: 

350 raise SystemError("Checking sizeof (%s) failed !" % 'PY_LONG_LONG') 

351 

352 res = config_cmd.check_type_size('long long', 

353 expected=expected['long long']) 

354 if res >= 0: 

355 #private_defines.append(('SIZEOF_%s' % sym2def('long long'), '%d' % res)) 

356 public_defines.append(('NPY_SIZEOF_%s' % sym2def('long long'), '%d' % res)) 

357 else: 

358 raise SystemError("Checking sizeof (%s) failed !" % 'long long') 

359 

360 if not config_cmd.check_decl('CHAR_BIT', headers=['Python.h']): 

361 raise RuntimeError( 

362 "Config wo CHAR_BIT is not supported" 

363 ", please contact the maintainers") 

364 

365 return private_defines, public_defines
